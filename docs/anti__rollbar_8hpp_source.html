69 void createAntiRollbar(JSON data, Vec3 scale, MobilizedBody &chassis_body, MobilizedBody &rocker_body_left, MobilizedBody &rocker_body_right)
70 {
71 auto arb_bend = GetVec3(data["antirollbar"]["arb"]["bend"], scale);
72 auto arb_middle = GetVec3(data["antirollbar"]["arb"]["center"], scale);
73
74 auto arb_droplink = GetVec3(data["antirollbar"]["droplink"]["arb"], scale);
75 auto droplink_rocker = GetVec3(data["antirollbar"]["droplink"]["rocker"], scale);
76
77 auto arb_rot_dir = arb_bend - arb_middle;
78 auto arb_len = arb_rot_dir.norm() * 2.0;
79
80 // Describe mass and visualization properties for a generic body.
81 auto arb_mass_props = GetMassInertia(data["antirollbar"]["arb"], 1.0, 1.0e-6);
82 Body::Rigid arbInfo(arb_mass_props);
83 arbInfo.addDecoration(Transform(FromDirectionVector(Vec3(0.0, 1.0, 0.0), XAxis)), DecorativeCylinder(0.01, arb_len / 2.0));
84
85 // Attach a revolute mobilizer to the chassis, this will assure a rotational degree of freedom of the ARB w.r.t. the chassis
86 m_arb_left = MobilizedBody::Revolute(chassis_body, TransformWorldToBody(chassis_body, arb_middle, arb_rot_dir), arbInfo, Transform());
87
88 // Attach the right hand side body (revolute w.r.t. right hand side body)
89 m_arb_right = MobilizedBody::Revolute(m_arb_left, TransformWorldToBody(m_arb_left, arb_middle, arb_rot_dir), arbInfo, Transform());
90
91 // Attach the droplink from the rocker body onto to the arb body
92 CreateLink(m_arb_left, arb_droplink, rocker_body_left, droplink_rocker);
93 CreateLink(m_arb_right, arb_droplink.elementwiseMultiply(Vec3(1.0, -1.0, 1.0)), rocker_body_right, droplink_rocker.elementwiseMultiply(Vec3(1.0, -1.0, 1.0)));
94 }
