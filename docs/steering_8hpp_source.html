85 void createSteeringSystem(JSON data, Vec3 scale, MobilizedBody &m_chassis, bool simplified = true)
86 {
87 /*
88 Steering System Overview:
89
90 | steering column ---------> |
91 | O------| <-- steering wheel
92 | intermediate shaft --> / |
93 | O
94 | \ / <-- steering shaft
95 | steering pinion \ /
96 | O
97 | \
98 | steering rack --> \
99 */
100
101 /*
102 The steering system consists of:
103 - steering rack (slider): lateral w.r.t. chassis
104 - pinion (1d velocity constraint): translation to rotational w.r.t. rack
105 - steering shaft (revolute): fixed to pinion and rotates w.r.t. chassis
106 - intermediate shaft (universal): with respect to steering shaft
107 - steering column (universal):
108
109 This part is not really ideal in terms of computational efficiency.
110 It probaby makes sense to model up to the rack mounted revolute joint and
111 lump the rest of the steering system inertia/mass alltogether in a single shaft.
112 */
113
114 // Take the tierod inner location as the end point of the steering rack
115 // with steering rack position in the middle of the chassis
116 auto steering_rack_end = GetVec3(data["suspension"]["toe_link"]["chassis"], scale);
117 auto steering_rack_pos = steering_rack_end.elementwiseMultiply(Vec3(1.0, 0.0, 1.0));
118
119 // Calculate steering rack direction and length
120 auto steering_rack_dir = (steering_rack_end - steering_rack_pos);
121 auto steering_rack_len = steering_rack_dir.norm() * 2.0;
122
123 // Describe mass and visualization properties for a generic body.
124 auto rack_mass_props = GetMassInertia(data["steering"]["steering_rack"], 1.0, 1.0e-6);
125 Body::Rigid steeringRackInfo(rack_mass_props);
126 steeringRackInfo.addDecoration(Transform(FromDirectionVector(Vec3(1.0, 0, 0), YAxis)), DecorativeCylinder(0.01, steering_rack_len / 2.0));
127
128 // A slider is defined along the common X-axis, redirect it along the steering axis
129 m_steering_rack = MobilizedBody::Slider(m_chassis, TransformWorldToBody(m_chassis, steering_rack_pos, steering_rack_dir, XAxis), steeringRackInfo, Transform(Vec3(0)));
130
131 // Get steering shaft and column coordinates
132 auto pinion_center_at_rack = GetVec3(data["steering"]["pinion"]["pos"], scale);
133 auto intermediate_shaft_forward = GetVec3(data["steering"]["intermediate_shaft"]["joint"], scale);
134 auto intermediate_shaft_rear = GetVec3(data["steering"]["steering_colum"]["joint"], scale);
135 auto steeringwheel_center = GetVec3(data["steering"]["steering_wheel"]["pos"], scale);
136
137 auto seering_column_dir = steeringwheel_center - intermediate_shaft_rear;
138
139 // Calculate virtual contact point to achieve desired steering ration
140 // auto steering_rack_ratio = 0.0024; // [rev/mm]
141 // auto effective_pinion_radius = (1.0 / steering_rack_ratio) * 1.0 / (2.0 * Pi) / 1000.0;
142 auto steering_rack_ratio = data["steering"]["pinion"]["ratio"].ToFloat(); // 15.0; // [rad/m]
143 auto effective_pinion_radius = (1.0 / steering_rack_ratio);
144
145 // Steering shaft and column mobilizers
146 // All joints are defined at the base for convenience (expressed in world frame coordinates converted to body local)
147
148 // Steering shaft
149 auto steering_shaft_dir = intermediate_shaft_forward - pinion_center_at_rack;
150 auto steering_shaft_len = steering_shaft_dir.norm();
151
152 auto shaft_mass_props = GetMassInertia(data["steering"]["steering_shaft"], 1.0, 1.0e-6);
153 Body::Rigid steeringShaftInfo(shaft_mass_props);
154 steeringShaftInfo.addDecoration(Transform(FromDirectionVector(Vec3(0.0, 1.0, 0), XAxis), Vec3(0.0)), DecorativeCylinder(0.01, steering_shaft_len / 2.0));
155
156 m_steering_shaft = MobilizedBody::Revolute(m_chassis, TransformWorldToBody(m_chassis, pinion_center_at_rack, steering_shaft_dir), steeringShaftInfo, Transform(Vec3(0.0, 0.0, -steering_shaft_len / 2.0)));
157
158 // Project a point at the center of the steering rack onto the steering shaft to find the virtual perpendicular contact point
159 auto pinion_virtual_cen = ProjectPointOnLine(steering_rack_pos, pinion_center_at_rack, intermediate_shaft_forward);
160 auto pinion_virtual_contact_dir = steering_rack_pos - pinion_virtual_cen;
161 // auto pinion_virtual_contact_pos = UnitVec3(pinion_virtual_contact_dir) * effective_pinion_radius + pinion_virtual_cen;
162 auto pinion_virtual_contact_pos = UnitVec3(pinion_virtual_contact_dir) * effective_pinion_radius + pinion_center_at_rack;
163
164 // Add a no slip 1d constraints to constrain the rotation, we need to use the virtual contact position here to get the correct steering ratio
165 Constraint::NoSlip1D(m_chassis, PosWorldToBody(m_chassis, pinion_virtual_contact_pos), UnitVec3(steering_rack_dir), m_steering_rack, m_steering_shaft);
166
167 // Induce artificial motion
168 Motion::Sinusoid(m_steering_shaft, Motion::Level::Position, 20.0 * Pi / 180.0, 2.0 * 2.0 * Pi, 0.0);
169
170 if (simplified)
171 {
172 // nothing further...
173 return;
174 }
175
176 // Intermediate shaft
177 auto intermediate_shaft_dir = intermediate_shaft_rear - intermediate_shaft_forward;
178 auto intermediate_shaft_len = intermediate_shaft_dir.norm();
179
180 auto intermediate_mass_props = GetMassInertia(data["steering"]["intermediate_shaft"], 1.0, 1.0e-6);
181 Body::Rigid intermediateShaftInfo(intermediate_mass_props);
182 intermediateShaftInfo.addDecoration(Transform(FromDirectionVector(Vec3(0.0, 1.0, 0), XAxis), Vec3(0.0)), DecorativeCylinder(0.01, intermediate_shaft_len / 2.0));
183
184 auto intermediate_shaft = MobilizedBody::Universal(m_steering_shaft, TransformWorldToBody(m_steering_shaft, intermediate_shaft_forward, intermediate_shaft_dir), intermediateShaftInfo, Transform(Vec3(0.0, 0.0, -intermediate_shaft_len / 2.0)));
185
186 // Steering column
187 auto steering_columm_dir = steeringwheel_center - intermediate_shaft_rear;
188 auto steering_column_len = steering_columm_dir.norm();
189
190 // Split the body mass and inertia in half, because we need both a chassis mounted and intermediate shaft mounted part which are welded together
191 auto column_mass_props = GetMassInertia(data["steering"]["steering_column"], 1.0 * 0.5, 1.0e-6 * 0.5);
192 Body::Rigid steeringColumnInfo(column_mass_props);
193
194 auto steering_column = MobilizedBody::Universal(intermediate_shaft, TransformWorldToBody(intermediate_shaft, intermediate_shaft_rear, steering_columm_dir), steeringColumnInfo, Transform(Vec3(0.0, 0.0, -steering_column_len / 2.0)));
195 steeringColumnInfo.addDecoration(Transform(FromDirectionVector(Vec3(0.0, 1.0, 0), XAxis), Vec3(0.0)), DecorativeCylinder(0.01, steering_column_len / 2.0));
196 steeringColumnInfo.addDecoration(Transform(FromDirectionVector(Vec3(0.0, 1.0, 0), XAxis), Vec3(0.0, 0.0, steering_column_len / 2.0)), DecorativeBrick(Vec3(0.125, 0.025, 0.05)));
197 auto steering_column_cha = MobilizedBody::Revolute(m_chassis, TransformWorldToBody(m_chassis, intermediate_shaft_rear, steering_columm_dir, ZAxis), steeringColumnInfo, Transform(Vec3(0.0, 0.0, -steering_column_len / 2.0)));
198
199 // Weld revolute chassis mounted joint to steering column
200 Constraint::Weld(steering_column_cha, steering_column);
201 /* */
202 }
203};
