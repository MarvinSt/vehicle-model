156 void createRocker(JSON data, Vec3 scale, MobilizedBody &m_chassis)

158 Body::Rigid rockerInfo(MassProperties(0.1, Vec3(0), UnitInertia(0.001)));

180 void createMultiLink(JSON data, Vec3 scale, MobilizedBody &m_chassis, MobilizedBody &m_steering)

182 /*
183 | ^ y+

193 | | ^

198 Vec3 chassis_link_pos;

202 upright_link_pos = GetVec3(data["lca_outer"], scale);
